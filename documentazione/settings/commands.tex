\let\oldchapter\chapter
\renewcommand{\chapter}[1]{
    \newpage
    \begingroup%
    \let\clearpage\relax% Stop LaTeX from going to a new page
    \vspace*{\fill}%
    \oldchapter{#1}
    \vspace*{\fill}%
    \endgroup
    \thispagestyle{empty}
    \newpage
}
